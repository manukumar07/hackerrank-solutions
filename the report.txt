SELECT
    CASE
        WHEN Grade >= 8 AND Marks BETWEEN Min_Mark AND Max_Mark THEN Name
        ELSE NULL
    END Name,
    Grade,
    Marks
FROM
    students s, grades g
WHERE
    Marks BETWEEN Min_Mark AND Max_Mark
ORDER BY
    Grade DESC,
    CASE
        WHEN Grade >= 8 AND Marks BETWEEN Min_Mark AND Max_Mark THEN Name
        ELSE NULL
    END,
    Marks
